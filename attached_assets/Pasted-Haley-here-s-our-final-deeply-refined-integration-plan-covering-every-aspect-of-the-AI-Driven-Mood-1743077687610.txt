## **1. Backend Deep Integration**

### **A. Enhanced Mood Capsule API Endpoint**

**Objective:**  
Process user content, detect layered emotions, group content items, and generate narrative captions with flexible tone options (Poetic, Concise, Conversational).

**Key Enhancements:**
- **Layered Sentiment Analysis:**  
  Detect if an item expresses multiple emotions (e.g., joy mixed with nostalgia) and assign weighted markers.
- **Metadata Tagging:**  
  Enrich each content item with timestamps, engagement metrics, and additional user-provided notes for context.
- **Dynamic Tone Options:**  
  Enable caption generation in multiple styles. For example, the same grouped summary can yield:
  - *Poetic:* “A burst of joyful energy that carries you forward…”
  - *Concise:* “A moment of pure joy.”
  - *Conversational:* “This really lit up your day!”

**Backend Code Example (Node.js/Express – Place in `server/api/moodCapsule.js`):**

```js
// server/api/moodCapsule.js
const express = require('express');
const axios = require('axios');
const router = express.Router();

/* 
  Helper: Simulate detailed sentiment analysis.
  In production, integrate with Hugging Face or OpenAI's sentiment analysis API.
*/
async function analyzeContent(contentItems) {
  return contentItems.map(item => ({
    ...item,
    // Assigning a weighted mood marker for demonstration.
    mood: item.data.includes('happy') ? 'joyful' : 'reflective',
    // Tag in metadata (simulate timestamp & tags)
    metadata: { timestamp: new Date().toISOString(), tags: ['demo'] }
  }));
}

/*
  Helper: Group the analyzed items.
  Incorporate options for “Hybrid Moods” if multiple emotions are detected.
*/
function groupContent(analyzedData, moodInput) {
  const group = {
    title: moodInput === 'joyful' ? 'Burst of Joy' : 'Moments of Reflection',
    subtitle: moodInput === 'joyful'
      ? 'Memories that illuminated your week.'
      : 'Thoughtful moments captured over time.',
    items: analyzedData.filter(item => item.mood === moodInput),
    summary: `A collection of ${moodInput} experiences`
  };
  return [group];
}

/*
  Helper: Generate a caption using GPT-based text generation.
  Replace YOUR_API_KEY with your actual key when integrating.
*/
async function generateCaption(summary, tone) {
  try {
    const response = await axios.post('https://api.openai.com/v1/completions', {
      prompt: `Generate a ${tone} caption for: "${summary}"`,
      max_tokens: 50,
      model: 'text-davinci-003'
    }, {
      headers: { 'Authorization': `Bearer YOUR_API_KEY` }
    });
    return response.data.choices[0].text.trim();
  } catch (error) {
    console.error('Caption generation failed:', error);
    return 'A moment captured in time.';
  }
}

// Main API Endpoint
router.post('/mood-capsule', async (req, res) => {
  try {
    const userContent = req.body.content; // Array of creative content items.
    const moodInput = req.body.mood;        // e.g., "joyful" or "reflective"
    const tone = req.body.tone || 'Poetic';   // Default tone is Poetic

    // 1. Analyze content with detailed sentiment detection.
    const analyzedData = await analyzeContent(userContent);
    
    // 2. Group content based on detected emotions (allows for hybrid moods).
    const groupedData = groupContent(analyzedData, moodInput);
    
    // 3. Generate a caption for each content group.
    const moodCapsules = await Promise.all(groupedData.map(async group => {
       const caption = await generateCaption(group.summary, tone);
       // Optionally, generate a mood-based color based on input (for frontend styling).
       const moodColor = moodInput === 'joyful' ? '#FFD700' : '#87CEEB';
       return {
         title: group.title,
         subtitle: group.subtitle,
         caption,
         tone,
         moodColor,   // Pass this to control capsule card appearance.
         items: group.items
       };
    }));

    // 4. (Optional) Trigger evolution point rewards here.

    res.json({ moodCapsules });
  } catch (e) {
    console.error(e);
    res.status(500).json({ error: 'Internal Server Error' });
  }
});

module.exports = router;
```

```tsx
// client/src/components/dashboard/MoodCapsules.tsx
import React, { useState, useEffect } from 'react';
import axios from 'axios';

export default function MoodCapsules() {
  const [capsules, setCapsules] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);

  // Function to generate mood capsules.
  const generateCapsule = async () => {
    setLoading(true);
    try {
      // Replace with actual user content-fetching logic.
      const userContent = [
        { id: 1, thumbnail: '/assets/image1.jpg', data: 'happy day at the beach' },
        { id: 2, thumbnail: '/assets/image2.jpg', data: 'sunset with friends' }
      ];
      const mood = "joyful";  // This can come from user input or mood-tracking
      const tone = "Poetic";  // Use tone preference if available
      
      const response = await axios.post('/api/mood-capsule', {
        content: userContent,
        mood,
        tone
      });
      setCapsules(response.data.moodCapsules);
    } catch (error) {
      console.error(error);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    generateCapsule();
  }, []);

  if (loading) return <div>Gathering your creative moments...</div>;

  return (
    <div className="mood-capsules-container" style={{ display: 'flex', flexDirection: 'column', gap: '1rem' }}>
      {capsules.map((capsule, index) => (
        <div
          key={index}
          className="capsule-card"
          style={{
            border: `3px solid ${capsule.moodColor}`,
            padding: '1rem',
            borderRadius: '8px',
            transition: 'transform 0.3s',
            cursor: 'pointer'
          }}
          onMouseEnter={e => (e.currentTarget.style.transform = 'scale(1.02)')}
          onMouseLeave={e => (e.currentTarget.style.transform = 'scale(1)')}
        >
          <h3>{capsule.title}</h3>
          <p>{capsule.subtitle}</p>
          <blockquote style={{ fontStyle: 'italic' }}>{capsule.caption}</blockquote>
          <div className="capsule-items" style={{ display: 'flex', gap: '0.5rem', marginTop: '0.5rem' }}>
            {capsule.items.map((item: any, i: number) => (
              <div key={i} style={{ position: 'relative' }}>
                <img 
                  src={item.thumbnail} 
                  alt="creative snippet"
                  style={{ width: '60px', height: '60px', objectFit: 'cover', borderRadius: '4px' }} 
                />
                <div
                  className="metadata-bubble"
                  style={{
                    position: 'absolute',
                    top: '-25px',
                    left: '0',
                    backgroundColor: 'rgba(0,0,0,0.6)',
                    color: 'white',
                    padding: '2px 6px',
                    borderRadius: '4px',
                    fontSize: '10px',
                    display: 'none'
                  }}
                >
                  {item.metadata.timestamp}
                </div>
              </div>
            ))}
          </div>
          <div className="capsule-actions" style={{ marginTop: '1rem', display: 'flex', gap: '0.5rem' }}>
            <button>Edit Capsule</button>
            <button>Share</button>
            <button>Refresh Caption</button>
          </div>
        </div>
      ))}
    </div>
  );
}
```
