# Mood Capsules Implementation for Creative Symbiosis

**Overview:**  
This feature dynamically groups user content based on detected emotional signals and generates mood-based captions. It leverages AI for sentiment analysis and captioning, grouping similar items into “Mood Capsules.” Users can view, tweak, and share these mood capsules—all without overthinking their emotional expression. The implementation includes backend endpoints (Node.js/Express) and a React component to display mood capsules.

---

## **1. Backend Enhancements**

### A. API Endpoint: `/api/mood-capsule`

Create a file (e.g., `server/api/moodCapsule.js`) in your Replit project. This endpoint processes user content, performs emotional analysis, groups items, and generates a caption for each group.

```js
// File: server/api/moodCapsule.js
const express = require('express');
const router = express.Router();

// Dummy helper functions – replace these with calls to your chosen AI/sentiment services.
async function analyzeContent(contentItems) {
  // Example: Return each item with a "mood" property.
  return contentItems.map(item => ({ ...item, mood: "joyful" }));
}

function groupContent(analyzedData, moodInput) {
  // Example: Group items with the same mood.
  const groups = [];
  const group = {
    title: "Moments of Joy",
    subtitle: "A collection of happy memories",
    items: analyzedData.filter(item => item.mood === moodInput),
    // A summary could be a simple aggregate string
    summary: "A burst of joyful energy"
  };
  groups.push(group);
  return groups;
}

async function generateCaption(summary, moodInput) {
  // Example: Use a simple template. Replace with an AI-generated call.
  return `Reflecting a feeling of ${moodInput}: ${summary}`;
}

// POST /api/mood-capsule
router.post('/mood-capsule', async (req, res) => {
  try {
    const { content, mood } = req.body;  // Expecting an array of content items and a mood string
    if (!content || !mood) {
      return res.status(400).json({ error: 'Content and mood are required.' });
    }
    
    // Step 1: Analyze content sentiment
    const analyzedData = await analyzeContent(content);
    
    // Step 2: Group content by mood markers
    const groupedData = groupContent(analyzedData, mood);
    
    // Step 3: Generate caption for each group
    const moodCapsules = await Promise.all(groupedData.map(async group => {
       const caption = await generateCaption(group.summary, mood);
       return {
         title: group.title,
         subtitle: group.subtitle,
         caption,
         items: group.items
       }
    }));
    
    // Optionally: Integrate with evolution points / creative history logging here.
    res.json({ moodCapsules });
  } catch (e) {
    console.error(e);
    res.status(500).json({ error: 'Internal Server Error' });
  }
});

module.exports = router;
```

**Instructions:**
- Install any required npm packages (e.g., `express`).
- Replace the dummy helper functions with your integrated AI/sentiment analysis services.
- Ensure this route is imported and used in your main server file (e.g., `server.js`).

## **2. Frontend Enhancements**

### A. React Component: `MoodCapsules.tsx`

Create a new file under `client/src/components/dashboard/MoodCapsules.tsx`. This component will query the API, display mood capsules as cards, and allow minimal inline editing.

```tsx
// File: client/src/components/dashboard/MoodCapsules.tsx
import React, { useState, useEffect } from 'react';
import axios from 'axios';

interface ContentItem {
  thumbnail: string;
  // Include additional fields as necessary.
}

interface MoodCapsule {
  title: string;
  subtitle: string;
  caption: string;
  items: ContentItem[];
}

export default function MoodCapsules() {
  const [capsules, setCapsules] = useState<MoodCapsule[]>([]);
  const [loading, setLoading] = useState(true);
  const mood = "joyful"; // This can later be dynamic based on user input or mood settings

  const generateCapsule = async () => {
    setLoading(true);
    try {
      // Replace the static content with real user content fetch logic
      const dummyContent = [
        { thumbnail: 'https://via.placeholder.com/100', id: 1 },
        { thumbnail: 'https://via.placeholder.com/100', id: 2 },
      ];
      const response = await axios.post('/api/mood-capsule', {
        content: dummyContent,
        mood
      });
      setCapsules(response.data.moodCapsules);
    } catch (e) {
      console.error('Error generating capsule:', e);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    generateCapsule();
  }, []);

  if (loading) return <div>Gathering your creative moments...</div>;

  return (
    <div className="mood-capsules">
      {capsules.map((capsule, index) => (
        <div 
          key={index} 
          className="capsule-card" 
          style={{ border: '2px solid #FFD700', marginBottom: '1rem', padding: '1rem' }}
        >
          <h3>{capsule.title}</h3>
          <p>{capsule.subtitle}</p>
          <blockquote>{capsule.caption}</blockquote>
          <div className="capsule-items" style={{ display: 'flex', gap: '0.5rem', flexWrap: 'wrap' }}>
            {capsule.items.map((item, i) => (
              <img 
                key={i} 
                src={item.thumbnail} 
                alt="Creative snippet" 
                style={{ width: '100px', height: '100px', objectFit: 'cover' }} 
              />
            ))}
          </div>
          <button onClick={() => {/* Open detailed view or edit mode */}} style={{ marginTop: '0.5rem' }}>
            Dive In
          </button>
        </div>
      ))}
      <button onClick={generateCapsule} style={{ marginTop: '1rem' }}>
        Generate New Capsule
      </button>
    </div>
  );
}
```

**Instructions:**
- Ensure Axios is installed (`npm install axios`).
- Modify dummy content fetching to integrate with your actual data source.
- Style this component using your existing CSS or styled-components to match your overall theme and mood-based color palettes.
