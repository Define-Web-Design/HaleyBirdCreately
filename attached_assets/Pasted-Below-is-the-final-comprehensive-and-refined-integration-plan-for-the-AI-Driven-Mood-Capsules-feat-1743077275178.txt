## **1. Backend Enhancements**

### **A. Mood Capsule API Endpoint**

Create a new endpoint `/api/mood-capsule` to process user content, analyze mood via sentiment detection, group related content, and generate a narrative caption.

**File:** `server/api/moodCapsule.js`

```js
// server/api/moodCapsule.js
const express = require('express');
const router = express.Router();

// Dummy helper functions for illustration.
// Replace these with actual integration logic (e.g., Hugging Face models or OpenAI API calls).
async function analyzeContent(contentItems) {
  // Example: Return the content with dummy "mood" markers
  return contentItems.map(item => ({ ...item, mood: 'joyful' }));
}

function groupContent(analyzedData, moodInput) {
  // Example: Group items based on the input mood filter (simplified)
  const group = {
    title: moodInput === 'joyful' ? 'Moments of Joy' : 'Reflective Moments',
    subtitle: moodInput === 'joyful' ? 'A collection of happy memories.' : 'Thoughtful and serene moments.',
    items: analyzedData.filter(item => item.mood === moodInput),
    summary: `A series of ${moodInput} experiences`
  };
  return [group];
}

async function generateCaption(summary, moodInput) {
  // Example: Call an AI text-generation API; here, we return a hardcoded caption.
  return moodInput === 'joyful'
    ? "A burst of joyful energy that lights up your day."
    : "A serene reflection of your inner thoughts.";
}

router.post('/mood-capsule', async (req, res) => {
  try {
    const userContent = req.body.content; // Expect an array of content items
    const moodInput = req.body.mood; // e.g., "joyful", "reflective", etc.
    
    // Step 1: Analyze content with sentiment/mood detection.
    const analyzedData = await analyzeContent(userContent);
    
    // Step 2: Group content based on detected mood markers.
    const groupedData = groupContent(analyzedData, moodInput);
    
    // Step 3: Generate a caption for each content group.
    const moodCapsules = await Promise.all(groupedData.map(async group => {
       const caption = await generateCaption(group.summary, moodInput);
       return {
         title: group.title,
         subtitle: group.subtitle,
         caption,
         items: group.items // Each item should include necessary data (e.g., thumbnail URL).
       }
    }));
    
    // Optionally: Award evolution points or update creative history here.
    res.json({ moodCapsules });
  } catch (e) {
    console.error(e);
    res.status(500).json({ error: 'Internal Server Error' });
  }
});

module.exports = router;
```

## **2. Frontend Enhancements**

### **A. Mood Capsules React Component**

Develop a new React component to display the Mood Capsules on your dashboard.

**File:** `client/src/components/dashboard/MoodCapsules.tsx`

```tsx
// client/src/components/dashboard/MoodCapsules.tsx
import React, { useState, useEffect } from 'react';
import axios from 'axios';

export default function MoodCapsules() {
  const [capsules, setCapsules] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);
  
  // Function to trigger Mood Capsule generation.
  const generateCapsule = async () => {
    setLoading(true);
    try {
      // Replace this with your method to fetch user's content.
      const userContent = [
        { id: 1, thumbnail: '/path/to/image1.jpg', data: 'Content 1' },
        { id: 2, thumbnail: '/path/to/image2.jpg', data: 'Content 2' }
      ];
      const mood = "joyful";  // This can be dynamic based on user input.
      
      const response = await axios.post('/api/mood-capsule', {
        content: userContent,
        mood
      });
      setCapsules(response.data.moodCapsules);
    } catch (e) {
      console.error(e);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    generateCapsule();
  }, []);

  if (loading) return <div>Gathering your creative moments...</div>;

  return (
    <div className="mood-capsules">
      {capsules.map((capsule, index) => (
        <div
          key={index}
          className="capsule-card"
          style={{ border: `2px solid #FFD700`, margin: '1rem', padding: '1rem', borderRadius: '8px' }}
        >
          <h3>{capsule.title}</h3>
          <p>{capsule.subtitle}</p>
          <blockquote>{capsule.caption}</blockquote>
          <div className="capsule-items" style={{ display: 'flex', gap: '0.5rem' }}>
            {capsule.items.map((item: any, i: number) => (
              <img key={i} src={item.thumbnail} alt="creative snippet" style={{ width: '60px', height: '60px', objectFit: 'cover', borderRadius: '4px' }} />
            ))}
          </div>
          <button onClick={() => {/* open capsule details/edit mode */}} style={{ marginTop: '1rem' }}>
            Dive In
          </button>
        </div>
      ))}
    </div>
  );
}
```
