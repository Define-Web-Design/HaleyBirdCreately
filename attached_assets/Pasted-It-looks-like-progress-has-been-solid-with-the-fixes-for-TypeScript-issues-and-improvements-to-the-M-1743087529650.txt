#### **Code Snippet Example:**
Here’s a starting point for integrating the MultiSelect component:
```tsx
import React, { useState } from 'react';
import MultiSelect from 'react-multi-select-component';

const MoodCapsuleForm = () => {
  const [selectedMoods, setSelectedMoods] = useState([]);
  
  const moodsOptions = [
    { label: "Joyful", value: "joyful" },
    { label: "Calm", value: "calm" },
    { label: "Nostalgic", value: "nostalgic" },
    { label: "Energetic", value: "energetic" },
  ];

  const handleSubmit = () => {
    if (!selectedMoods.length) {
      alert("Please select at least one mood.");
      return;
    }
    // Submit logic
    console.log("Submitting: ", selectedMoods);
  };

  return (
    <div>
      <h3>Create Mood Capsule</h3>
      <label htmlFor="moods">Select Moods:</label>
      <MultiSelect
        options={moodsOptions}
        value={selectedMoods}
        onChange={setSelectedMoods}
        labelledBy="Select Moods"
      />
      <button onClick={handleSubmit}>Submit Capsule</button>
    </div>
  );
};

export default MoodCapsuleForm;
```
